public class Main {

	public static void main(String[] args) {

		for(int i=1; i<10; i++){
			if(i==3 || i==5){
				continue;
			}
			for(int j=1; j<10; j++){
				if((i*j)/10==0){
					System.out.print("  "+i*j);
				}else{
					System.out.print(" "+i*j);
				}
			}
		System.out.println();
		}

	}

}

import java.util.Random;
import java.util.Scanner;

public class Main{
    public static void main(String[] args){
    	Scanner scan=new Scanner(System.in);
    	Random rnd= new Random();

    	int N =rnd.nextInt(99)+1;
    	int M =0;

    	for(int i=0;i<10;i++){
	    	System.out.println("1～100の数値を入力：");
	    	M=scan.nextInt();

	    	if(M<N){
	    		System.out.println(M+"より大きい値です");
	    	}else if(M>N){
	    		System.out.println(M+"より小さい値です ");
	    	}else{
	    		System.out.println("おめでとう。"+i+"回で当たりました。");
	    		break;
	    	}
	    	if(i==9){
	    		System.out.println("残念 です。10回推理しても当たりませんでした");
	    	}
    	}
    	scan.close();
    }
}
