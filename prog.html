import java.util.Scanner;

public class Kadai1180612 {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);

		int point = 0;
		int N =0;

		System.out.println("点数入力:");
		point = scan.nextInt();

		if(point<40){
			N = 40-point;
			System.out.println("不可Bです、あと"+N+"点で不可Ａです");
		}else if(point<60){
			N = 60-point;
			System.out.println("不可Ａです、あと"+N+"点で可です");
		}else if(point<65){
			N = 65-point;
			System.out.println("可です、あと"+N+"点で良です");
		}else if(point<80){
			N = 80-point;
			System.out.println("良です、あと"+N+"点で優です");
		}else{
			System.out.println("優です");
		}

		scan.close();
	}

}

import java.util.Scanner;

public class Kadai4180612 {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);

		int week = 0;
		int time = 0;

		System.out.println("曜日選択:");
		System.out.println("0=日曜、1=月曜、2=火曜、3=水曜、4=木曜、5=金曜、6=土曜:");
		week = scan.nextInt();

		System.out.println("時間帯選択:");
		System.out.println("0=午前、1=午後、2=夜間:");
		time = scan.nextInt();

		if(week == 0){
			System.out.println("休診です");
		}else if((week == 1) || (week == 4)){
			System.out.println("診療しています");
		}else if((week == 2) || (week == 5)){
			if(time == 0){
				System.out.println("休診です");
			}else{
				System.out.println("診療しています");
			}
		}else if(week == 3){
			if(time == 2){
				System.out.println("休診です");
			}else{
				System.out.println("診療しています");
			}
		}else{
			if(time == 0){
				System.out.println("診療しています");
			}else{
				System.out.println("休診です");
			}
		}
		scan.close();
	}

}

import java.util.Random;
import java.util.Scanner;

public class Kadai5180612 {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		Random rnd= new Random();

		int N=rnd.nextInt(2);
		int P=0;

		System.out.println("プレイヤの手(1:ぐう 2:ちょき  3:ぱあ)：");
		P = scan.nextInt();

		if(P==1) {
			if(N==0) {
				System.out.println("プレイヤの手はぐう、ボスの手はぐうで引き分け");
			}else if(N==1) {
				System.out.println("プレイヤの手はぐう、ボスの手はちょきでプレイヤの勝ち");
			}else {
				System.out.println("プレイヤの手はぐう、ボスの手はぱあでプレイヤの負け");
			}
		}

		if(P==2) {
			if(N==0) {
				System.out.println("プレイヤの手はちょき、ボスの手はぐうでプレイヤの負け");
			}else if(N==1) {
				System.out.println("プレイヤの手はちょき、ボスの手はちょきで引き分け");
			}else {
				System.out.println("プレイヤの手はちょき、ボスの手はぱあでプレイヤの勝ち");
			}
		}

		if(P==3) {
			if(N==0) {
				System.out.println("プレイヤの手はぱあ、ボスの手はぐうでプレイヤの勝ち");
			}else if(N==1) {
				System.out.println("プレイヤの手はぱあ、ボスの手はちょきでプレイヤの負け");
			}else {
				System.out.println("プレイヤの手はぱあ、ボスの手はぱあで引き分け");
			}
		}

		scan.close();

	}

}
