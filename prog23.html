                import java.util.Scanner;

                public class Main2302 {
                
                    public static void main(String[] args) {
                        Scanner scan=new Scanner(System.in);
                        String str=scan.next();
                        int n=str.length();
                        char ch;
                        int b=0;
                        for(int i=0;i<n;i++) {//A=65,a=97,Z=90
                            ch=str.charAt(i);
                            b=ch;
                            if(b<=90) {
                                b+=32;
                                ch=(char)b;
                            }else {
                                b-=32;
                                ch=(char)b;
                            }
                            System.out.print(ch);
                        }
                
                    }
                
                }

                import java.util.Scanner;

                        while(true) {
                            n=scan.nextInt();
                            if(str.length()<c+n) {
                                b[i]=str.substring(c,str.length());
                                i++;
                                break;
                            }else {
                                b[i]=str.substring(c,c+n);
                            }
                            i++;
                            c+=n;
                        }
                        for(int j=0;j<i;j++) {
                            System.out.println(b[j]);
                        }
                
                    }
