public class Kadai1180529 {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		
		int N = 0;
		
		System.out.println("数値入力：");
		N = scan.nextInt();
		
		if(N<10) {
			System.out.println("入力され"+N+"は10未満の整数です。");
		}else {
			System.out.println("入力された"+N+"は10以上の整数です。");
		}
		
		scan.close();

	}

}

import java.util.Scanner;

public class Kadai2180529 {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		
		int N = 0;
		double sum = 0;
		
		System.out.println("人数入力：");
		N = scan.nextInt();
		
		if(5<=N) {
			sum = 850 * N * 0.7;
			System.out.println("入館料は850×"+N+"人x0.7="+sum+"円です。");
		}else {
			sum = 850 * N;
			System.out.println("入館料は850×"+N+"人="+sum+"円です。");
		}
		
		scan.nextInt();

	}

}
