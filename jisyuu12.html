import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scan=new Scanner(System.in);
        double n = 10000;
        int year = 0;
        while ( n < 20000 ){
            n *= 1.05;
            year++;
        }
        System.out.println(year+"Years Later");
        scan.close();
    }
}

import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scan=new Scanner(System.in);
        int N = scan.nextInt();
        int i = 1;
        int sum=0;

        while ( i <= N ){
            if(i%7==0){
                sum+=i;
            }
            i++;
        }
        System.out.println("sum = "+sum);
        scan.close();
    }
}
