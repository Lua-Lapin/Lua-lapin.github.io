    public static void main(String[] args){

        int sum = 0, i; 
        
        for (i = 1; ; i++){

        	}

        } 
        System.out.println("sum = " + sum);
        stdIn.close();
    }

import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scan=new Scanner(System.in);

        int N =0;
        double total=0;
        int totali=0;
        double ave =0;
        int data =0;
        int notdata =0;

        while(total<100){
        	N = scan.nextInt();
        	if(N>=0){
        		totali += N;
        		total += N;
        		data++;
        	}else{
        		notdata++;
        	}

        }
        ave = total/data;
        System.out.println("DataNum="+data);
        System.out.println("InvalidDataNum="+notdata);
        System.out.println("Total="+totali);
        System.out.printf("Average="+"%.1f",ave);
        scan.close();
    }
}
